1. Get the application URL by running these commands:
{{- if .Values.ragApp.ingress.enabled }}
  https://{{ (index .Values.ragApp.ingress.hosts 0).host }}
{{- else if contains "NodePort" .Values.ragApp.service.type }}
  export NODE_PORT=$(kubectl get --namespace {{ .Release.Namespace }} -o jsonpath="{.spec.ports[0].nodePort}" services {{ include "rag-app-slack.fullname" . }})
  export NODE_IP=$(kubectl get nodes --namespace {{ .Release.Namespace }} -o jsonpath="{.items[0].status.addresses[0].address}")
  echo http://$NODE_IP:$NODE_PORT
{{- else if contains "LoadBalancer" .Values.ragApp.service.type }}
  NOTE: It may take a few minutes for the LoadBalancer IP to be available.
  You can watch the status of by running 'kubectl get --namespace {{ .Release.Namespace }} svc -w {{ include "rag-app-slack.fullname" . }}'
  export SERVICE_IP=$(kubectl get svc --namespace {{ .Release.Namespace }} {{ include "rag-app-slack.fullname" . }} --template "{{"{{ range (index .status.loadBalancer.ingress 0) }}{{.}}{{ end }}"}}")
  echo http://$SERVICE_IP:{{ .Values.ragApp.service.port }}
{{- else if contains "ClusterIP" .Values.ragApp.service.type }}
  export POD_NAME=$(kubectl get pods --namespace {{ .Release.Namespace }} -l "app.kubernetes.io/name={{ include "rag-app-slack.name" . }},app.kubernetes.io/instance={{ .Release.Name }}" -o jsonpath="{.items[0].metadata.name}")
  export CONTAINER_PORT=$(kubectl get pod --namespace {{ .Release.Namespace }} $POD_NAME -o jsonpath="{.spec.containers[0].ports[0].containerPort}")
  echo "Visit http://127.0.0.1:8080 to use your application"
  kubectl --namespace {{ .Release.Namespace }} port-forward $POD_NAME 8080:$CONTAINER_PORT
{{- end }}

2. Check deployment status:
  kubectl get pods --namespace {{ .Release.Namespace }} -l "app.kubernetes.io/name={{ include "rag-app-slack.name" . }},app.kubernetes.io/instance={{ .Release.Name }}"

3. View logs:
  kubectl logs --namespace {{ .Release.Namespace }} -l "app.kubernetes.io/name={{ include "rag-app-slack.name" . }},app.kubernetes.io/instance={{ .Release.Name }}" -f

4. Access Slack endpoints:
  - Health: /api/slack/stats
  - Channels: /api/slack/channels
  - Import: POST /api/slack/import
  - Search: POST /api/slack/search
  - Threads: GET /api/slack/threads/{id}

5. Verify Slack integration is working:
  kubectl exec --namespace {{ .Release.Namespace }} -it deployment/{{ include "rag-app-slack.fullname" . }} -- curl http://localhost:{{ .Values.ragApp.service.targetPort }}/api/slack/stats
